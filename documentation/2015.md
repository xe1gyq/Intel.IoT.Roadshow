### WiFi

    root@ubilinux:/home/edison# nano /etc/network/interfaces
    
    # interfaces(5) file used by ifup(8) and ifdown(8)
    auto lo
    iface lo inet loopback

    #auto usb0
    #iface usb0 inet static
    #    address 192.168.2.15
    #    netmask 255.255.255.0
    
    auto wlan0
    iface wlan0 inet dhcp
        # For WPA
        #wpa-ssid "itesm"
        #wpa-psk ""
        # For WEP
        wireless-essid itesm
        wireless-mode Managed
        wireless-key s:""
    
    root@ubilinux:/home/edison# ifup wlan0
    root@ubilinux:/home/edison# reboot

### Cmake

    root@ubilinux:~$ cd
    root@ubilinux:~$ wget http://www.cmake.org/files/v3.2/cmake-3.2.2.tar.gz
    root@ubilinux:~$ tar xvf cmake-3.2.2.tar.gz
    root@ubilinux:~$ cd cmake-3.2.2
    root@ubilinux:~/cmake-3.2.2$ ./bootstrap
    root@ubilinux:~/cmake-3.2.2$ make
    root@ubilinux:~/cmake-3.2.2$ su
    root@ubilinux:~/cmake-3.2.2# make install
